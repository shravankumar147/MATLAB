1function [y,pauseSim,stopSim] = HelperAudioToneRemovalProcessing(x,args,param)
 2%HELPERAUDIOTONEREMOVALPROCESSING Main processing for audio tone removal.
 3%
 4% This function HelperAudioToneRemovalProcessing is only in support of 
 5% AudioInterferingToneRemovalExample. It may change in a future release.
 6
 7% Copyright 2013 The MathWorks, Inc.
 8
 9
10Fs   = args.SampleRate;
11
12persistent hn
13if isempty(hn)  
14    hn = dsp.NotchPeakFilter('SampleRate',Fs,'CenterFrequency',...
15        param(1).InitialValue,'Bandwidth',param(2).InitialValue);
16end
17
18% Obtain new values for parameters through UDP Receive
19[paramNew, simControlFlags] = HelperUnpackUDP();
20
21y=0;
22pauseSim = simControlFlags.pauseSim;
23stopSim = simControlFlags.stopSim;
24
25if simControlFlags.stopSim
26    return;  % Stop the simulation
27end
28if simControlFlags.pauseSim
29    return; % Pause the simulation (but keep checking for commands from GUI)
30end
31
32u = randn(size(x,1),1);
33y = step(hn,[x,u]);
34
35if ~isempty(paramNew)   % If tuning hasn't started
36    if simControlFlags.resetObj   % If "Reset" button is pressed
37        reset(hn);
38    else
39        hn.CenterFrequency = paramNew(1);
40        hn.Bandwidth       = paramNew(2);        
41    end
42end
43
44
45
46
