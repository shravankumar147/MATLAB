1function [paramNew, simControlFlags] = HelperUnpackUDP
  2%HELPERUNPACKUDP Unpack contents of UDP packet from tuning UI
  3% Function to resolve the UDP packet received from Parameter Tuning UI
  4% into parameter values, reset flag and stop simulation flag.
  5% Input:
  6%         none
  7% Output:
  8%         paramNew        - Array of parameter values received from the UI
  9%         simControlFlags - Structure containing flags for simulation
 10%                           control. These include the following:
 11%                           resetObj: Flag to reset the System object. 
 12%                                     If true, the object needs to be reset
 13%                           pauseSim: Flag to pause the simulation. If 
 14%                                     true, the simulation needs to be 
 15%                                     paused.
 16%                           stopSim:  Flag to stop the simulation. If true,
 17%                                     the simulation needs to be stopped
 18%
 19% This function HelperUnpackUDP is only in support of
 20% HelperCreateParamTuningUI. It may change in a future release.
 21%
 22% Example:
 23%         % obj is a System object with a tunable parameter 'Position'
 24%         obj = dsp.TimeScope('Position', [100 100 560 420]);
 25%         
 26%         % Define parameters to be tuned
 27%         param(1).Name = 'X-Position';
 28%         param(1).InitialValue = obj.Position(1);
 29%         param(1).Limits = [0, 400];
 30% 
 31%         % Create UI to tune the Position parameter
 32%         hUI = HelperCreateParamTuningUI(param);
 33% 
 34%         while(1)  % Continue until 'Stop Simulation' button is pressed
 35%             [paramNew, simControlFlags] = HelperUnpackUDP(); 
 36%                    % Obtain new values for parameter through UDP Receive
 37% 
 38%             if simControlFlags.stopSim      % Check if simulation needs to be stopped
 39%                 break;
 40%             end
 41%             if simControlFlags.pauseSim
 42%                 drawnow;
 43%                 continue;
 44%             end
 45%             if ~isempty(paramNew)
 46%                 if simControlFlags.resetObj     % Check if object needs to be reset
 47%                     reset(obj);
 48%                 end
 49% 
 50%                 obj.Position(1) = paramNew(1);  % Change Parameter value
 51%             end
 52% 
 53%             % Now that the parameter is tuned, call the STEP function
 54%             x = randn;
 55%             step(obj, x);
 56%         end
 57%       
 58%         % Cleanup
 59%         close(hUI);      % Close the tuning UI
 60%         release(obj);     % Release the System object
 61
 62% Copyright 2013 The MathWorks, Inc.
 63
 64persistent hUDPReceiver resetSwitch pauseSwitch
 65
 66if isempty(hUDPReceiver)
 67    % UDP Receiver
 68    hUDPReceiver = dsp.UDPReceiver('MessageDataType','double',...
 69                            'ReceiveBufferSize',400);
 70    resetSwitch = false;
 71    pauseSwitch = false;
 72end
 73
 74simControlFlags.resetObj = false;
 75simControlFlags.pauseSim = pauseSwitch;
 76simControlFlags.stopSim  = false;
 77
 78paramNew = step(hUDPReceiver);
 79if isempty(paramNew)
 80    return;
 81else
 82    % paramNew has new values for parameters for the first N elements,
 83    % where N is the number of parameters tuned in the UI. The last three
 84    % elements correspond to the Reset, Pause Simulation and Stop
 85    % Simulation buttons.
 86    if length(paramNew) < 4  % check for sporadic UDP input from other sources
 87        paramNew = [];
 88        return;
 89    end
 90    if paramNew(end-2) ~= resetSwitch  % Reset System object
 91        simControlFlags.resetObj = true;
 92        resetSwitch = ~resetSwitch;
 93    end
 94    if paramNew(end-1) ~= pauseSwitch  % Pause simulation
 95        pauseSwitch = ~pauseSwitch;
 96        simControlFlags.pauseSim = pauseSwitch;
 97    end
 98    if paramNew(end)                   % Stop simulation
 99        simControlFlags.stopSim = true;
100        release(hUDPReceiver);
101    end
102    paramNew = paramNew(1:end-3, 1);   % Parameter values
103end
